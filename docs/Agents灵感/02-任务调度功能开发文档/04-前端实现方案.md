## 2. schedulerStore 设计

```typescript
// src/stores/schedulerStore.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { invoke } from '@tauri-apps/api/core'
import { listen } from '@tauri-apps/api/event'

export interface QueuedTask {
  subtaskId: number
  todoId: number
  title: string
  priority: number
  enqueuedAt: number
  status: string
}

export interface TaskDependency {
  id: number
  subtaskId: number
  dependsOnId: number
  dependencyType: 'finish-to-start' | 'finish-to-finish' | 'start-to-start'
}

export interface SchedulerConfig {
  maxGlobalConcurrent: number
  maxProjectConcurrent: number
  maxQueueSize: number
  queueTimeoutSecs: number
}

export interface ScheduleStatus {
  queueLength: number
  runningCount: number
  maxConcurrent: number
  tasks: QueuedTask[]
}

export const useSchedulerStore = defineStore('scheduler', () => {
  const queueStatus = ref<ScheduleStatus | null>(null)
  const config = ref<SchedulerConfig | null>(null)
  const dependencies = ref<Map<number, TaskDependency[]>>(new Map())

  // 获取队列状态
  async function refreshQueueStatus() {
    queueStatus.value = await invoke<ScheduleStatus>('get_queue_status')
  }

  // 子任务入队
  async function enqueueSubtask(subtaskId: number) {
    await invoke('enqueue_subtask', { subtaskId })
    await refreshQueueStatus()
  }

  // 子任务出队
  async function dequeueSubtask(subtaskId: number) {
    await invoke('dequeue_subtask', { subtaskId })
    await refreshQueueStatus()
  }

  // 添加依赖
  async function addDependency(
    subtaskId: number,
    dependsOnId: number,
    depType: string = 'finish-to-start',
  ) {
    await invoke('add_dependency', { subtaskId, dependsOnId, depType })
  }

  // 删除依赖
  async function removeDependency(dependencyId: number) {
    await invoke('remove_dependency', { dependencyId })
  }

  // 获取依赖关系
  async function loadDependencies(subtaskId: number) {
    const deps = await invoke<TaskDependency[]>('get_dependencies', { subtaskId })
    dependencies.value.set(subtaskId, deps)
  }

  // 获取配置
  async function loadConfig() {
    config.value = await invoke<SchedulerConfig>('get_scheduler_config')
  }

  // 更新配置
  async function updateConfig(newConfig: SchedulerConfig) {
    await invoke('update_scheduler_config', { config: newConfig })
    config.value = newConfig
  }

  // 监听调度事件
  async function subscribeEvents() {
    await listen('scheduler:queue-updated', () => {
      refreshQueueStatus()
    })

    await listen<{ subtaskId: number; status: string }>('scheduler:task-status', (event) => {
      // 更新子任务调度状态
    })
  }

  return {
    queueStatus,
    config,
    dependencies,
    refreshQueueStatus,
    enqueueSubtask,
    dequeueSubtask,
    addDependency,
    removeDependency,
    loadDependencies,
    loadConfig,
    updateConfig,
    subscribeEvents,
  }
})
```

## 3. UI 组件设计

### 3.1 SchedulerPanel（调度控制面板）

嵌入在 EditorView 的子任务区域，提供调度相关的控制。

**功能：**
- 显示当前 Todo 的调度策略（manual/auto/scheduled）
- 切换调度策略
- 配置 Cron 表达式（当策略为 scheduled 时）
- 显示队列中的任务数量
- 启用/禁用自动调度

**位置：** EditorView 中，子任务列表的上方或 Agent 配置按钮附近

```vue
<template>
```

### 3.3 调度状态标签颜色映射

```typescript
function scheduleTagType(status: string): string {
  const map: Record<string, string> = {
    draft: 'info',
    pending: 'warning',
    queued: '',
    running: 'primary',
    completed: 'success',
    failed: 'danger',
    reviewing: 'warning',
    approved: 'success',
    rejected: 'danger',
    cancelled: 'info',
  }
  return map[status] || 'info'
}

function scheduleStatusLabel(status: string): string {
  const map: Record<string, string> = {
    none: '未调度',
    draft: '草稿',
    pending: '待调度',
    queued: '排队中',
    running: '执行中',
    completed: '已完成',
    failed: '失败',
    reviewing: '待审核',
    approved: '已通过',
    rejected: '已拒绝',
    cancelled: '已取消',
  }
  return map[status] || status
}
```

### 3.4 DependencyGraph（依赖关系图）

简单的 SVG 流程图，展示子任务间的依赖关系。

**功能：**
- 节点表示子任务，颜色表示调度状态
- 箭头表示依赖关系
- 点击节点可跳转到子任务编辑
- 高亮当前可执行的节点（依赖已满足）
- 支持添加/删除依赖（拖拽或弹框）

**实现思路：**
- 使用简单的分层布局算法（拓扑排序 + 层级分配）
- 小规模子任务（通常 < 20 个）不需要复杂的力导向布局
- 使用 SVG 绘制，不引入额外的图表库

```vue
<template>
  <div class="dependency-graph">
    <svg :width="graphWidth" :height="graphHeight">
      <!-- 连线 -->
      <g v-for="edge in edges" :key="edge.id">
        <line
          :x1="edge.from.x" :y1="edge.from.y"
          :x2="edge.to.x" :y2="edge.to.y"
          stroke="#94a3b8"
          stroke-width="2"
          marker-end="url(#arrowhead)"
        />
      </g>

      <!-- 节点 -->
      <g v-for="node in nodes" :key="node.id"
         @click="onNodeClick(node)"
         class="graph-node"
      >
        <rect
          :x="node.x - 60" :y="node.y - 20"
          width="120" height="40"
          rx="8" ry="8"
          :fill="nodeColor(node.status)"
          stroke="#e2e8f0"
        />
        <text
          :x="node.x" :y="node.y + 4"
          text-anchor="middle"
          font-size="12"
          fill="white"
        >
          {{ truncate(node.title, 10) }}
        </text>
      </g>

      <!-- 箭头标记 -->
      <defs>
        <marker id="arrowhead" markerWidth="10" markerHeight="7"
                refX="10" refY="3.5" orient="auto">
          <polygon points="0 0, 10 3.5, 0 7" fill="#94a3b8" />
        </marker>
      </defs>
    </svg>
  </div>
</template>
```

## 4. 定时任务管理界面

### 4.1 入口位置

在主界面（App.vue 或设置页面）中增加"定时任务"入口，可以通过以下方式之一：

- **方案 A**：在设置面板中新增"定时任务"标签页
- **方案 B**：在主界面底部工具栏增加"定时任务"按钮，打开独立面板
- **方案 C**：在 EditorView 的 Agent 配置弹窗中增加"定时调度"区域

推荐 **方案 A**，在设置面板中管理所有定时任务，提供统一的视角。
